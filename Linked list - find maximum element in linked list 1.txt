Linked list - find maximum element in linked list 1
Write a Program to find the maximum number in a list.

Input Format

Read elements upto -1

Constraints

1<=n<=10^7

Output Format

print output

Sample Input 0

1
5
2
4
7
3
-1
Sample Output 0

7
Sample Input 1

6
3
8
2
9
1
6
10
-1
Sample Output 1

10
class Node:
    def __init__(self,x):
        self.data=x
        self.next=None
class LinkedList:
    def __init__(self):
        self.head=self.tail=None
    def create(self,x):
        n=Node(x)
        if self.head==None:
            self.head=self.tail=n
        else:
            self.tail.next=n
            self.tail=n
    def show(self,n):
        temp=self.head
        while temp:
            if temp.data==n:
                print(n,"is present in the list")
                return
            else:
                temp=temp.next
        if temp==None:
            print(n,"not present in the list")
l1=LinkedList()
while True:
    x=int(input())
    if x==-1:
        n=int(input())
        break
    else:
        l1.create(x)
l1.show(n)
